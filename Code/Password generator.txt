#include<bits/stdc++.h>
using namespace std;

void password(int len)
{
    char str[] = "abcdefghijklmnopqrstuvwxyzABCD"
                 "EFGHIJKLMNOPQRSTUVWXYZ0123456789 ";
    int ind;
    // string pass;
    fstream f;
    f.open("Pass.txt",ios::out);
    for(int i=0; i<=len; i++)
    {
        ind=rand()%63+1;           ///63-- total char
        f<<str[ind];
    }
    f.close();
}

int main()
{
    int length;
    cout<<"Enter your password length : ";  ///At most 29
    cin>>length;
    password(length);
}
