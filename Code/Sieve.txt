#include <bits/stdc++.h>
#define ll long long
#define max 100000007
using namespace std;

bool is_prime[max]= {0};

void sieve(ll n)
{
    is_prime[1]=1;
    for(ll i=4; i<=n; i+=2)
    {
        is_prime[i]=1; //not prime
    }
    ll sq=sqrt(n)-1;
    for(ll i=3; i<=sq; i+=2)
    {
        if(is_prime[i]==0)
        {
            for(ll j=i*i; j<=n; j+=i)
                is_prime[j]=1;
        }
    }
}

int main()
{
    ll n;
    cin>>n;
    sieve(n);
    for(ll i=2; i<n; i++)
    {
        if(is_prime[i]==0)
            cout<<i<<" ";
    }
    cout<<endl;
}
